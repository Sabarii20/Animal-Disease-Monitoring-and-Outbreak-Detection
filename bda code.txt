import json
import threading
import pandas as pd
import joblib
from kafka import KafkaConsumer
from kafka.errors import NoBrokersAvailable
import streamlit as st
import plotly.express as px
import time
from queue import Queue

# --- 1. Page Configuration ---
st.set_page_config(page_title="Animal Disease Outbreak Dashboard", layout="wide")
st.title("🐾 Real-Time Animal Disease Severity Dashboard")
st.markdown("Use the filters in the sidebar to analyze data by state and species.")

# --- 2. Load the ML Model and Encoders ---
@st.cache_resource
def load_resources():
    try:
        model = joblib.load("severity_model.pkl")
        feature_encoders = joblib.load("severity_feature_encoders.pkl")
        target_encoder = joblib.load("severity_target_encoder.pkl")
        return model, feature_encoders, target_encoder
    except FileNotFoundError:
        st.sidebar.error("ERROR: Model/encoder files not found.")
        st.sidebar.info("Please run the `train_severity_model.py` script first to generate the necessary .pkl files.")
        return None, None, None

model, feature_encoders, target_encoder = load_resources()

if not all([model, feature_encoders, target_encoder]):
    st.stop()
else:
    st.sidebar.success("Severity model and encoders loaded successfully.")

# --- 3. Kafka Consumer Setup ---
@st.cache_resource
def get_kafka_consumer():
    try:
        consumer = KafkaConsumer(
            'animal_events',
            bootstrap_servers='localhost:9092',
            auto_offset_reset='earliest',
            value_deserializer=lambda v: json.loads(v.decode('utf-8', 'ignore')),
            consumer_timeout_ms=1000
        )
        st.sidebar.info("Successfully connected to Kafka.")
        return consumer
    except NoBrokersAvailable:
        st.sidebar.error("Kafka Connection Error: Could not connect to the broker at 'localhost:9092'.")
        return None

consumer = get_kafka_consumer()
if not consumer:
    st.stop()

# --- 4. Prediction Function ---
def predict_severity(event):
    """Processes a Kafka message and predicts the outbreak severity."""
    try:
        species = event.get("animal", {}).get("species", "Unknown")
        location = event.get("receiver", {}).get("state", "Unknown")
        symptoms = ", ".join(sorted([r.get("veddra_term_name", "Unknown") for r in event.get("reaction", [])]))
        outcome = event.get("outcome", [{}])[0].get("medical_status", "Unknown")

        # Placeholder fields
        disease = "Not specified"
        temperature = 39.0  # average animal temperature

        data_for_prediction = {
            "species": [species], "location": [location], "symptoms": [symptoms],
            "disease": [disease], "temperature": [temperature], "outcome": [outcome]
        }
        df = pd.DataFrame(data_for_prediction)

        # Encode categorical features
        for col, encoder in feature_encoders.items():
            if col in df.columns:
                df[col] = df[col].astype(str)
                df[col] = df[col].apply(lambda s: encoder.transform([s])[0] if s in encoder.classes_ else -1)

        expected_model_features = ['species', 'location', 'symptoms', 'disease', 'temperature', 'outcome']
        df = df.reindex(columns=expected_model_features, fill_value=-1)

        prediction_code = model.predict(df)[0]
        severity_level = target_encoder.inverse_transform([prediction_code])[0]

        return [species, location, symptoms, outcome, severity_level]
    except Exception as e:
        print(f"Error during prediction: {e}")
        return ["Error", "Error", "Error", "Error", "Error"]

# --- 5. Initialize Session State and Data Queue ---
if 'data_df' not in st.session_state:
    st.session_state.data_df = pd.DataFrame(columns=["Species", "Location", "Symptoms", "Outcome", "Severity"])
if 'consumer_thread_started' not in st.session_state:
    st.session_state.consumer_thread_started = False
if 'data_queue' not in st.session_state:
    st.session_state.data_queue = Queue()

# --- 6. Background Thread for Kafka Consumer ---
def consume_kafka(q):
    """Continuously consumes messages from Kafka and puts results in a queue."""
    while True:
        try:
            # Poll messages in a safe, non-blocking way
            raw_msgs = consumer.poll(timeout_ms=1000)
            for _, messages in raw_msgs.items():
                for msg in messages:
                    prediction_result = predict_severity(msg.value)
                    q.put(prediction_result)
            time.sleep(1)  # small sleep to avoid busy-wait
        except Exception as e:
            print(f"CRITICAL ERROR IN CONSUMER THREAD: {e}")
            time.sleep(5)

if not st.session_state.consumer_thread_started:
    threading.Thread(target=consume_kafka, args=(st.session_state.data_queue,), daemon=True).start()
    st.session_state.consumer_thread_started = True

# --- 7. Dashboard Rendering Logic ---

# Get all new data from the queue and append it to the main DataFrame
new_data_list = []
while not st.session_state.data_queue.empty():
    new_data_list.append(st.session_state.data_queue.get())

if new_data_list:
    new_df = pd.DataFrame(new_data_list, columns=["Species", "Location", "Symptoms", "Outcome", "Severity"])
    st.session_state.data_df = pd.concat([st.session_state.data_df, new_df], ignore_index=True)

df = st.session_state.data_df
